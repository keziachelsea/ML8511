        const int uvPin = A0; // Analog pin connected to the OUT pin of ML8511
        const int refPin = 7; // Optional, digital pin to control the enable (EN) pin of ML8511 (if available)

        void setup() {
        Serial.begin(9600); // Initialize serial communication
        pinMode(refPin, OUTPUT); // Set the enable pin as output (if used)
        digitalWrite(refPin, HIGH); // Enable the sensor (if used)
        }

        void loop() {
        // Read the analog value from the UV sensor
        int uvAnalogValue = analogRead(uvPin);
        // Convert the analog value to voltage
        float voltage = uvAnalogValue * (5.0 / 1023.0);

        // Convert the voltage to a UV index value (calibrated for ML8511)
        // Note: Adjust these thresholds based on your calibration and setup
        float uvIndex = (voltage - 1.0) * 15.0; // Subtract offset voltage and scale
        // Clamp the UV index to prevent negative or extreme values
        if (uvIndex < 0) uvIndex = 0;
        if (uvIndex > 15) uvIndex = 15;

        // Print the results
        Serial.print("UV Analog Value: ");
        Serial.print(uvAnalogValue);
        Serial.print("\tVoltage: ");
        Serial.print(voltage, 2);
        Serial.print(" V\tUV Index: ");
        Serial.println(uvIndex);

        delay(1000); // Delay for 1 second
        }
